let cpuScore = 0;
let userScore = 0;

// A function that randomly returns rock, paper, or scissors.
function getComputerChoice() {
    const cpuChoice = Math.floor(Math.random() * 10);

    if (cpuChoice <= 3) {
        console.log("COMPUTER: I Choose Rock!");
        return "rock";
    } else if (cpuChoice > 3 && cpuChoice <= 6) {
        console.log("COMPUTER: I Choose Paper!");
        return "paper";
    } else {
        console.log("COMPUTER: I Choose Scissors!");
        return "scissors";
    }
}

// Get user input with prompt
function getUserChoice() {
    let userChoice = prompt('Choose your hand!', 'Rock, Paper, Or Scissors?');
    userChoice = userChoice.toLowerCase();

    if (userChoice === 'rock') {
        console.log("HUMAN: I Choose Rock!");
    } else if (userChoice === 'paper') {
        console.log("HUMAN: I Choose Paper!");
    } else {
        console.log("HUMAN: I Choose Scissors!");
    }

    return userChoice;
}

// A function that takes the players' choices as arguments
// Plays a single round, increments the winner's score, and logs the result
function playRound(userChoice, cpuChoice) {
    if (userChoice === cpuChoice) {
        console.log('Draw Game!');
    } else if (
        (userChoice === "rock" && cpuChoice === "scissors") ||
        (userChoice === "paper" && cpuChoice === "rock") ||
        (userChoice === "scissors" && cpuChoice === "paper")
    ) {
        console.log(`${userChoice.charAt(0).toUpperCase() + userChoice.slice(1)} beats ${cpuChoice}!`);
        console.log('Winner is HUMAN!');
        userScore += 1;
    } else {
        console.log(`${cpuChoice.charAt(0).toUpperCase() + cpuChoice.slice(1)} beats ${userChoice}!`);
        console.log('Winner is COMPUTER!');
        cpuScore += 1;
    }
}

const humanHand = getUserChoice();
const cpuHand = getComputerChoice();

playRound(humanHand, cpuHand);

console.log('Score is');
console.log(`HUMAN : ${userScore} || ${cpuScore} : COMPUTER`);
